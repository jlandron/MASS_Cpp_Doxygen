87  int map(int initPopulation, vector<int> size, vector<int> index,
88  Place *curPlace) {
89  // compute the total # places
90  int placeTotal = 1;
91  for (int x = 0; x < int(size.size()); x++) placeTotal *= size[x];
92 
93  // compute the global linear index
94  int linearIndex = 0;
95  for (int i = 0; i < int(index.size()); i++) {
96  if (index[i] >= 0 && size[i] > 0 && index[i] < size[i]) {
97  linearIndex = linearIndex * size[i];
98  linearIndex += index[i];
99  }
100  }
101 
102  // compute #agents per place a.k.a. colonists
103  int colonists = initPopulation / placeTotal;
104  int remainders = initPopulation % placeTotal;
105  if (linearIndex < remainders) colonists++; // add a remainder
106 
107  return colonists;
108  };
